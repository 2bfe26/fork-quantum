### With Closing Disabled

```jsx
...
import { Modal, Button } from '@catho/quantum';

export default function ModalWithClosingDisabled() {
  const [openModal, setOpenModal] = useState(false);

  return (
    <>
      <Button onClick={() => setOpenModal(true)}>Click to open Modal</Button>
      {openModal && <Modal WithClosingDisabled >Content...</Modal>}
    </>
  );
}
```
